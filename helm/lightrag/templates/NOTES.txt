=============================================================================
  LightRAG - Retrieval-Augmented Generation with Knowledge Graphs
=============================================================================

To access your LightRAG deployment:

1. Get the route URL:

   {{- if .Values.route.enabled }}
   export ROUTE_URL=$(oc get route {{ include "lightrag.fullname" . }} -n {{ .Release.Namespace }} -o jsonpath='{.spec.host}')
   echo "LightRAG API: https://$ROUTE_URL"
   {{- else }}
   # Route is disabled. Access via port-forward:
   kubectl port-forward svc/{{ include "lightrag.fullname" . }} 9621:9621 -n {{ .Release.Namespace }}
   export ROUTE_URL="localhost:9621"
   {{- end }}

2. Test the health endpoint:

   curl https://$ROUTE_URL/health

3. Insert a document:

   curl -X POST https://$ROUTE_URL/insert \
     -H 'Content-Type: application/json' \
     -d '{"text": "Your document content here"}'

4. Query with different modes:

   # Hybrid mode (recommended)
   curl -X POST https://$ROUTE_URL/query \
     -H 'Content-Type: application/json' \
     -d '{"query": "Your question", "mode": "hybrid", "top_k": 10}'

   # Mix mode (combines KG + vector)
   curl -X POST https://$ROUTE_URL/query \
     -H 'Content-Type: application/json' \
     -d '{"query": "Your question", "mode": "mix", "top_k": 10}'
